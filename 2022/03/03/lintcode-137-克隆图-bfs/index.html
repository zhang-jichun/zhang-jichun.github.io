    nodeSet = self.getNodes(node)

    self.cloneLines(nodeSet, nodeDic)
    return nodeDic[node]

def getNodes(self, node):
    res = set([node])
    queue = [node]
    while queue:
        popNode = queue.pop(0)
        for temp in popNode.neighbors:
            if temp in res:
                continue
            res.add(temp)
            queue.append(temp)
    return res

def cloneNodes(self, nodeSet):
    res = {}
    for node in nodeSet:
        res[node] = UndirectedGraphNode(node.label)
    return res
